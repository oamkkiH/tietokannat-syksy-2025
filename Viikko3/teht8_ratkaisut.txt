-- a) Tulosta DVD-elokuvien kielet aakkosjärjestyksessä.
SELECT name
FROM language
ORDER BY name;

-- b) Tulosta kaikkien niiden näyttelijöiden elokuvat, joiden sukunimi on Temple.
SELECT film.title
FROM film
JOIN film_actor
  ON film.film_id = film_actor.film_id
JOIN actor
  ON film_actor.actor_id = actor.actor_id
WHERE actor.last_name = 'Temple';

-- c) Tulosta elokuvassa "Ghost Groundhog" näytelleet näyttelijät.
SELECT actor.first_name,
       actor.last_name
FROM actor
JOIN film_actor
  ON actor.actor_id = film_actor.actor_id
JOIN film
  ON film_actor.film_id = film.film_id
WHERE film.title = 'Ghost Groundhog';

-- d) Montako kauhuelokuvaa ("Horror") tietokannassa on?
SELECT COUNT(*)
FROM film_category
JOIN category
  ON film_category.category_id = category.category_id
WHERE category.name = 'Horror';

-- e) Tulosta kaikki kauhuelokuvat.
SELECT film.title
FROM film
JOIN film_category
  ON film.film_id = film_category.film_id
JOIN category
  ON film_category.category_id = category.category_id
WHERE category.name = 'Horror';
