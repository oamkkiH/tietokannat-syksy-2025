USE employees;

-- a) Tulosta kymmenen ensimmäistä työntekijää sukunimen mukaan.
SELECT * FROM employees ORDER BY last_name LIMIT 10;

-- b) Tulosta kymmenen ensimmäistä työntekijää sukunimen ja etunimen mukaan.
SELECT * FROM employees ORDER BY last_name, first_name LIMIT 10;

-- c) Tulosta viisi viimeksi palkattua työntekijää.
SELECT * FROM employees ORDER BY hire_date DESC LIMIT 5;

-- d) Suurin nykyinen palkka: etunimi, sukunimi ja palkka.
SELECT employees.first_name, employees.last_name, salaries.salary FROM employees JOIN salaries ON employees.emp_no = salaries.emp_no WHERE salaries.to_date = '9999-01-01' AND salaries.salary = (SELECT MAX(salary) FROM salaries WHERE to_date = '9999-01-01');

-- e) Pienin nykyinen palkka: etunimi, sukunimi ja palkka.
SELECT employees.first_name, employees.last_name, salaries.salary FROM employees JOIN salaries ON employees.emp_no = salaries.emp_no WHERE salaries.to_date = '9999-01-01' AND salaries.salary = (SELECT MIN(salary) FROM salaries WHERE to_date = '9999-01-01');

-- f) Työntekijät, joiden nykyinen palkka ylittää 150000.
SELECT employees.first_name, employees.last_name, salaries.salary FROM employees JOIN salaries ON employees.emp_no = salaries.emp_no WHERE salaries.to_date = '9999-01-01' AND salaries.salary > 150000;

-- g) Kuinka monta työntekijää työskentelee Sales-osastossa?
SELECT COUNT(*) FROM dept_emp JOIN departments ON dept_emp.dept_no = departments.dept_no WHERE departments.dept_name = 'Sales' AND dept_emp.to_date = '9999-01-01';

-- g) Kuinka monta työntekijää työskentelee Marketing-osastossa?
SELECT COUNT(*) FROM dept_emp JOIN departments ON dept_emp.dept_no = departments.dept_no WHERE departments.dept_name = 'Marketing' AND dept_emp.to_date = '9999-01-01';

-- h) Osastonjohtajien etunimi, sukunimi ja osasto.
SELECT employees.first_name, employees.last_name, departments.dept_name FROM employees JOIN dept_manager ON employees.emp_no = dept_manager.emp_no JOIN departments ON dept_manager.dept_no = departments.dept_no WHERE dept_manager.to_date = '9999-01-01';

-- i) Myynnissä työskentelevien keskipalkka.
SELECT AVG(salaries.salary) FROM salaries JOIN dept_emp ON salaries.emp_no = dept_emp.emp_no JOIN departments ON dept_emp.dept_no = departments.dept_no WHERE salaries.to_date = '9999-01-01' AND dept_emp.to_date = '9999-01-01' AND departments.dept_name = 'Sales';

-- i) Markkinoinnissa työskentelevien keskipalkka.
SELECT AVG(salaries.salary) FROM salaries JOIN dept_emp ON salaries.emp_no = dept_emp.emp_no JOIN departments ON dept_emp.dept_no = departments.dept_no WHERE salaries.to_date = '9999-01-01' AND dept_emp.to_date = '9999-01-01' AND departments.dept_name = 'Marketing';
